# Two-Way LLM Routing Platform - Build Specification

## Executive Summary

This document provides complete specifications for building a production-ready LLM API routing platform based on `any-gateway`, extended with Stripe subscription billing, multi-provider routing with caching, and credit risk management. The initial focus is **Western customers accessing Chinese LLM providers** (DeepSeek, Qwen, Kimi, GLM), with empty interfaces for future China→Western expansion.

**Target output:** A working prototype deployable to Fly.io within 5-7 days by a 2-agent team using opencode/crush.

class CacheInterface(ABC):
    """Abstract interface for caching strategies"""
    
    @abstractmethod
    async def get(self, key: str) -> Optional[Dict]:
        """Retrieve from cache"""
        pass
    
    @abstractmethod

class SemanticCache(CacheInterface):
    """Semantic similarity cache using embeddings"""

        self.redis = redis_client
        self.threshold = similarity_threshold
    
    async def get(self, key: str) -> Optional[Dict]:
        # For semantic cache, key is an embedding vector
        # Simplified: use exact match for MVP
        data = await self.redis.get(f"cache:exact:{self._hash_key(key)}")

    async def set(self, key: str, value: Dict, ttl: int) -> None:
        await self.redis.setex(
            f"cache:exact:{self._hash_key(key)}",
            ttl,

    async def delete_pattern(self, pattern: str) -> int:
        keys = await self.redis.keys(f"cache:exact:{pattern}")

class BatchInterface(ABC):
    """Abstract interface for request batching"""
    
    @abstractmethod

class DynamicBatcher(BatchInterface):
    """Dynamic batch aggregator with max latency guarantee"""
    
    def __init__(self, redis_client: redis.Redis, 
                 max_batch_size: int = 32,
                 max_wait_ms: int = 100):
        self.redis = redis_client
        self.max_batch_size = max_batch_size
        self.max_wait_ms = max_wait_ms
        self.batches: Dict[str, List] = {}
        self.futures: Dict[str, List[asyncio.Future]] = {}
        self._flush_task = None
    
    async def start(self):
        """Start background flusher"""
        self._flush_task = asyncio.create_task(self._periodic_flush())
    
    async def add_request(self, request_id: str, request: RoutingRequest,
                          future: asyncio.Future) -> None:
        model_key = request.model
        if model_key not in self.batches:
            self.batches[model_key] = []
            self.futures[model_key] = []
        
        self.batches[model_key].append(request)
        self.futures[model_key].append(future)
        
        if len(self.batches[model_key]) >= self.max_batch_size:
            await self._flush_model(model_key)
    
    async def _periodic_flush(self):
        while True:
            await asyncio.sleep(self.max_wait_ms / 1000)
            for model_key in list(self.batches.keys()):
                if self.batches[model_key]:
                    await self._flush_model(model_key)
    
    async def _flush_model(self, model_key: str):

# Circuit Breaker
# ============================================================================

class CircuitBreaker:
    """Circuit breaker for provider failure handling"""
    
    def __init__(self, redis_client: redis.Redis, 
                 failure_threshold: int = 5,
                 recovery_timeout: int = 60,
                 half_open_max_calls: int = 3):
        self.redis = redis_client
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_max_calls = half_open_max_calls
    
    async def call(self, provider_name: str, func, *args, **kwargs):
        """Execute function with circuit breaker protection"""
        state = await self._get_state(provider_name)
        
        if state == ProviderStatus.OPEN:
            last_failure = await self._get_last_failure(provider_name)
            if last_failure and (datetime.now() - last_failure).seconds > self.recovery_timeout:
                await self._set_state(provider_name, ProviderStatus.HALF_OPEN)
                state = ProviderStatus.HALF_OPEN
            else:
                raise Exception(f"Circuit open for {provider_name}")
        
        try:
            result = await func(*args, **kwargs)
            
            if state == ProviderStatus.HALF_OPEN:
                await self._record_success(provider_name)
                if await self._get_success_count(provider_name) >= self.half_open_max_calls:
                    await self._reset(provider_name)
            
            await self._record_success(provider_name)
            return result
            
        except Exception as e:
            await self._record_failure(provider_name)
            
            failure_count = await self._get_failure_count(provider_name)
            if failure_count >= self.failure_threshold:
                await self._set_state(provider_name, ProviderStatus.OPEN)
            
            raise e
    
    async def _get_state(self, provider_name: str) -> ProviderStatus:
        state = await self.redis.get(f"cb:{provider_name}:state")
        return ProviderStatus(state.decode() if state else ProviderStatus.CLOSED)
    
    async def _set_state(self, provider_name: str, state: ProviderStatus):
        await self.redis.set(f"cb:{provider_name}:state", state.value)
    
    async def _record_failure(self, provider_name: str):
        await self.redis.incr(f"cb:{provider_name}:failures")
        await self.redis.setex(f"cb:{provider_name}:last_failure", 3600, datetime.now().isoformat())
    
    async def _record_success(self, provider_name: str):
        await self.redis.incr(f"cb:{provider_name}:successes")
    
    async def _get_failure_count(self, provider_name: str) -> int:
        val = await self.redis.get(f"cb:{provider_name}:failures")
        return int(val) if val else 0
    
    async def _get_success_count(self, provider_name: str) -> int:
        val = await self.redis.get(f"cb:{provider_name}:successes")
        return int(val) if val else 0
    
    async def _get_last_failure(self, provider_name: str) -> Optional[datetime]:
        val = await self.redis.get(f"cb:{provider_name}:last_failure")
        return datetime.fromisoformat(val.decode()) if val else None
    
    async def _reset(self, provider_name: str):
        await self.redis.delete(f"cb:{provider_name}:state")
        await self.redis.delete(f"cb:{provider_name}:failures")
        await self.redis.delete(f"cb:{provider_name}:successes")
        await self.redis.delete(f"cb:{provider_name}:last_failure")

# Main FastRouter
# ============================================================================

class FastRouter:
    """Main routing engine with caching, batching, circuit breakers"""
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.cache = SemanticCache(redis_client)
        self.batcher = DynamicBatcher(redis_client)
        self.circuit_breaker = CircuitBreaker(redis_client)
        self.providers: Dict[str, Provider] = {}
        
        # Start background tasks
        asyncio.create_task(self.batcher.start())
    
    def register_provider(self, provider: Provider):
        """Register a provider for routing"""
        self.providers[provider.name] = provider
    
    async def route(self, request: RoutingRequest) -> RoutingResponse:
        """Main entry point for request routing"""
        
        # 1. Check cache first
        cache_key = self._generate_cache_key(request)
        cached_response = await self.cache.get(cache_key)
        if cached_response:
            return RoutingResponse(**cached_response, cached=True)
        
        # 2. Select best provider
        provider = await self._select_provider(request)
        if not provider:
            raise Exception("No available provider")
        
        # 3. Call provider with circuit breaker
        response = await self.circuit_breaker.call(
            provider.name,
            self._call_provider,
            provider,
            request
        )
        
        # 4. Cache response if appropriate (not streaming)
        if not request.stream and response.content:
            await self.cache.set(cache_key, {
                "provider": provider.name,
                "model": response.model,
                "content": response.content,
                "usage": response.usage,
                "latency_ms": response.latency_ms
            }, 3600)
        
        return response
    
    async def _select_provider(self, request: RoutingRequest) -> Optional[Provider]:
        """Select the best provider based on strategy"""
        # Filter by model support
        eligible = [
            p for p in self.providers.values()
            if any(request.model.startswith(prefix) for prefix in p.models)
        ]
        
        if not eligible:
            return None
        
        if request.prefer_cost:
            # Cheapest provider (deepseek is cheapest)
            return min(eligible, key=lambda p: 1 if p.name == "deepseek" else 2)
        elif request.prefer_latency:
            # Closest geo from where request originated
            # Simplified: returns provider with highest weight
            return max(eligible, key=lambda p: p.weight)
        else:
            # Weighted round-robin
            return await self._weighted_round_robin(eligible)
    
    async def _weighted_round_robin(self, providers: List[Provider]) -> Provider:
        """Weighted round-robin selection"""
        # Simplified for MVP - just return first
        return providers[0]
    
    async def _call_provider(self, provider: Provider, request: RoutingRequest) -> RoutingResponse:
        """Call provider API with timeout and retry"""
        import aiohttp
        import time
        
        start_time = time.time()
        
        async with aiohttp.ClientSession() as session:
            payload = {
                "model": request.model,
                "messages": request.messages,
                "temperature": request.temperature,
                "max_tokens": request.max_tokens,
                "stream": request.stream
            }
            
            headers = {
                "Authorization": f"Bearer {provider.api_key}",
                "Content-Type": "application/json"
            }
            
            async with session.post(
                f"{provider.api_base}/v1/chat/completions",
                json=payload,
                headers=headers,
                timeout=aiohttp.ClientTimeout(total=provider.timeout)
            ) as resp:
                data = await resp.json()
                latency_ms = int((time.time() - start_time) * 1000)
                
                return RoutingResponse(
                    provider=provider.name,
                    model=data.get("model", request.model),
                    content=data["choices"][0]["message"]["content"],
                    usage=data.get("usage", {"prompt_tokens": 0, "completion_tokens": 0}),
                    latency_ms=latency_ms,
                    cached=False
                )
    
    def _generate_cache_key(self, request: RoutingRequest) -> str:
        """Generate cache key from request"""
        # Simplified: hash of messages and parameters
        key_data = {
            "messages": request.messages,
            "temperature": request.temperature,
            "max_tokens": request.max_tokens
        }
        return json.dumps(key_data, sort_keys=True)

from fastapi import Request, HTTPException
from starlette.middleware.base import BaseHTTPMiddleware
from typing import Optional, Dict, Any
import asyncio
from datetime import datetime

class UsageTrackingMiddleware(BaseHTTPMiddleware):
    """
    Middleware to track token usage and deduct from user balance.
    Uses async queue to avoid blocking requests.
    """
    
    def __init__(self, app, db_pool, redis_client):
        super().__init__(app)
        self.db = db_pool
        self.redis = redis_client
        self._deduction_queue = asyncio.Queue()
        self._worker_task = None
    
    async def dispatch(self, request: Request, call_next):
        # Check authentication
        api_key = request.headers.get("Authorization", "").replace("Bearer ", "")
        user_id = await self._get_user_from_api_key(api_key)
        
        if not user_id:
            raise HTTPException(status_code=401, detail="Invalid API key")
        
        # Check if user has sufficient balance/tokens
        has_quota = await self._check_quota(user_id)
        if not has_quota:
            raise HTTPException(status_code=402, detail="Insufficient balance")
        
        # Process request
        response = await call_next(request)
        
        # Extract usage from response body (requires response body capture)
        # For MVP, approximate usage from request/response
        usage = await self._extract_usage(request, response)
        
        # Queue deduction (non-blocking)
        await self._deduction_queue.put({
            "user_id": user_id,
            "tokens": usage.get("total_tokens", 0),
            "cost_usd": usage.get("cost_usd", 0),
            "model": usage.get("model", "unknown"),
            "provider": usage.get("provider", "unknown")
        })
        
        # Start worker if not running
        if not self._worker_task or self._worker_task.done():
            self._worker_task = asyncio.create_task(self._process_deductions())
        
        return response
    
    async def _get_user_from_api_key(self, api_key: str) -> Optional[int]:
        """Get user ID from API key (cached in Redis)"""
        # Try Redis cache first
        cached_user = await self.redis.get(f"apikey:{api_key}")
        if cached_user:
            return int(cached_user)
        
        # Fallback to database
        async with self.db.acquire() as conn:
            result = await conn.fetchrow(
                "SELECT user_id FROM api_keys WHERE key_hash = crypt($1, key_hash)",
                api_key
            )
            if result:
                user_id = result["user_id"]
                await self.redis.setex(f"apikey:{api_key}", 300, user_id)
                return user_id
        
        return None
    
    async def _check_quota(self, user_id: int) -> bool:
        """Check if user has sufficient quota/balance"""
        async with self.db.acquire() as conn:
            user = await conn.fetchrow(
                "SELECT quota_usd, subscription_tier, tokens_used_this_month, monthly_token_limit FROM users WHERE id = $1",
                user_id
            )
            
            if not user:
                return False
            
            # Check prepaid balance
            if user["quota_usd"] <= 0 and user["subscription_tier"] == "free":
                return False
            
            # Check monthly token limit for subscription users
            if user["subscription_tier"] != "free":
                if user["tokens_used_this_month"] >= user["monthly_token_limit"]:
                    # Check if overage is allowed
                    return False
            
            return True
    
    async def _extract_usage(self, request: Request, response) -> Dict[str, Any]:
        """
        Extract token usage from request/response.
        Extend for actual token counting.
        """
        # TODO: Implement actual token counting with tiktoken
        # For MVP, estimate 1 token per ~4 characters
        body = await request.body()
        content = body.decode("utf-8") if body else ""
        
        # Very rough estimate - replace with actual tokenizer
        estimated_tokens = len(content) // 4 if content else 100
        
        return {
            "total_tokens": estimated_tokens,
            "cost_usd": estimated_tokens * 0.000001,  # $0.000001 per token
            "model": request.query_params.get("model", "unknown"),
            "provider": "deepseek"  # TODO: Get from routing engine
        }
    
    async def _process_deductions(self):
        """Background worker to process usage deductions"""
        while True:
            try:
                item = await self._deduction_queue.get()
                
                async with self.db.acquire() as conn:
                    # Deduct from balance
                    await conn.execute(
                        """
                        UPDATE users 
                        SET quota_usd = quota_usd - $1,
                            tokens_used_this_month = tokens_used_this_month + $2
                        WHERE id = $3 AND quota_usd >= $1
                        """,
                        item["cost_usd"],
                        item["tokens"],
                        item["user_id"]
                    )
                    
                    # Record usage
                    await conn.execute(
                        """
                        INSERT INTO token_usage (user_id, provider, model, input_tokens, output_tokens, cost_usd, created_at)
                        VALUES ($1, $2, $3, $4, $5, $6, NOW())
                        """,
                        item["user_id"],
                        item["provider"],
                        item["model"],
                        item["tokens"] // 2,  # Rough split
                        item["tokens"] // 2,
                        item["cost_usd"]
                    )
                
                self._deduction_queue.task_done()
                
            except Exception as e:
                # Log error and continue
                print(f"Deduction failed: {e}")
                self._deduction_queue.task_done()
